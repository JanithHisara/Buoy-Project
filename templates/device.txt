    <script>
        async function loadPorts() {
            try {
                const response = await fetch('/ports');
                const ports = await response.json();
                const select = document.getElementById('ports');
                const currentValue = select.value;
                select.innerHTML = '';
                ports.forEach(port => {
                    const option = document.createElement('option');
                    option.value = port;
                    option.text = port;
                    if (port === currentValue) option.selected = true;
                    select.appendChild(option);
                });
            } catch (error) {
                console.error('Failed to load ports:', error);
            }
        }
        
        async function connectToDevice() {
            const port = document.getElementById('ports').value;
            try {
                const response = await fetch('/connect', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ port: port })
                });
                const data = await response.json();
                showNotif(data.message || `Connected to ${port}`);
            } catch (error) {
                showNotif(`Connect command sent to ${port}`);
            }
        }
        
        async function scanGPS() {
            try {
                await fetch('/scan');
                showNotif('GPS scan started — awaiting data');
            } catch (error) {
                showNotif('Scan sent (no server response)');
            }
        }
        
        window.onload = () => {
            loadPorts();
            setInterval(loadPorts, 5000);
        };
        let connectionType = 'serial';

        function toggleConnectionType() {
            const toggle = document.querySelector('.toggle-switch');
            const serialSection = document.getElementById('serial-section');
            const ethernetSection = document.getElementById('ethernet-section');

            const serialLabel = document.getElementById('serial-label');
            const ethernetLabel = document.getElementById('ethernet-label');

            connectionType = connectionType === 'serial' ? 'ethernet' : 'serial';

            toggle.classList.toggle('active');

            if (connectionType === 'ethernet') {
                serialSection.style.display = 'none';
                ethernetSection.style.display = 'flex';

                serialLabel.classList.remove('active');
                ethernetLabel.classList.add('active');
            } else {
                serialSection.style.display = 'flex';
                ethernetSection.style.display = 'none';

                serialLabel.classList.add('active');
                ethernetLabel.classList.remove('active');
            }
        }

        async function connectToDevice() {
            let payload = {};

            if (connectionType === 'serial') {
                payload = {
                    type: 'serial',
                    port: document.getElementById('ports').value
                };
            } else {
                const ip = document.getElementById('device-ip').value.trim();

                if (!ip) {
                    showNotif('Enter a valid IP address');
                    return;
                }

                payload = {
                    type: 'ethernet',
                    ip: ip
                };
            }

            try {
                const response = await fetch('/connect', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                const data = await response.json();
                showNotif(data.message || 'Connection successful');
            } catch (error) {
                showNotif('Connection request sent');
            }
        }
